417 /*! \brief QuickSort (recursive implementation)
418  \ingroup jkqtptools
419 
420  implementation from http://www.linux-related.de/index.html?/coding/sort/sort_quick.htm
421 */
422 template <class T>
423 inline void jkqtpQuicksort(T* a, int l, int r){
424  if(r>l){
425  int i=l-1;
426  int j=r;
427 
428  for(;;){
429  while(a[++i]<a[r]);
430  while(a[--j]>a[r] && j>i);
431  if(i>=j) break;
432  jkqtpSwap<T>(a, i, j);
433  }
434  jkqtpSwap<T>(a, i, r);
435 
436  jkqtpQuicksort(a, l, i-1);
437  jkqtpQuicksort(a, i+1, r);
438  }
439 }

442 /*! \brief QuickSort (recursive implementation), sorts \a a2 alongside \a a, using \a a as sort criterion
443  \ingroup jkqtptools
444 
445  implementation from http://www.linux-related.de/index.html?/coding/sort/sort_quick.htm
446 */
447 template <class T, class T2>
448 inline void jkqtpQuicksort(T* a, T2* a2, int l, int r){
449  if(r>l){
450  int i=l-1;
451  int j=r;
452 
453  for(;;){
454  while(a[++i]<a[r]);
455  while(a[--j]>a[r] && j>i);
456  if(i>=j) break;
457  jkqtpSwap(a, i, j);
458  jkqtpSwap(a2, i, j);
459  }
460  jkqtpSwap(a, i, r);
461  jkqtpSwap(a2, i, r);
462 
463  jkqtpQuicksort(a, a2, l, i-1);
464  jkqtpQuicksort(a, a2, i+1, r);
465  }
466 }

471 /*! \brief sort the given arrays, using \a input as sort criterion
472  \ingroup jkqtptools
473 
474  \param input array to be sorted
475  \param input2 array to be sorted
476  \param N size of the array input
477  \param output if \c !=nullptr data is written here (the memory location pointed at by \a output has to have at least the length \a N !!!),
478  otherwise the array input is sorted inplace.
479  \param output2 if \c !=nullptr data is written here (the memory location pointed at by \a output has to have at least the length \a N !!!),
480  otherwise the array input is sorted inplace.
481 
482  */
483 template <class T, class T2>
484 inline void jkqtpSort(T* input, T2* input2, int N, T* output=nullptr, T2* output2=nullptr) {
485  if ((!input)) return ;
486  if (N<=0) return;
487  T* data=input;
488  if (output!=nullptr) {
489  data=output;
490  memcpy(output, input, N*sizeof(T));
491  }
492  T2* data2=input2;
493  if (output2!=nullptr && input2!=nullptr) {
494  data2=output2;
495  memcpy(output2, input2, N*sizeof(T2));
496  }
497  jkqtpQuicksort(data, data2, 0, N-1);
498 }
