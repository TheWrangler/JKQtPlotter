39 /*! \brief This implements vertical <a href="http://en.wikipedia.org/wiki/Box_plot">boxplots</a>
40  \ingroup jkqtplotter_statgraphs
41 
42  The x position is given in posColumn. All other data are given in the medianColumn, minColumn, maxColumn,
43  percentile25Column and percentile75Column.
44 
45  \image html plot_boxplotvertical.png
46 
47  The different features of a boxplot are:
48 
49  \image html boxplots.png
50 
51  \note There are additional classes to draw a single boxplot element: JKQTPBoxplotHorizontalElement and JKQTPBoxplotVerticalElement.
52  In these you can set the data values, as they are NOT drawn from a data column. This can be useful, if you e.g. want to
53  draw the statistical properties of a distribution.
54 
55  \section BoxplotOutliers Outliers
56 
57  If you also want to display outliers, as shown here with circles:
58 
59  \image html boxplot_outliers.png
60 
61  You need to add them as (x,y)-pairs to the datastore and add a separate JKQTPXYLineGraph that shows these. See \ref JKQTPlotterBoxplotsGraphs for details. Here is an example code-snippet:
62 
63  \code
64  // 4. create a graph of vertical boxplots:
65  JKQTPBoxplotVerticalGraph* graph=new JKQTPBoxplotVerticalGraph(&plot);
66  graph->setPosColumn(columnPOS);
67  graph->setMinColumn(columnMIN);
68  graph->setPercentile25Column(columnQ25);
69  graph->setMedianColumn(columnMEDIAN);
70  graph->setMeanColumn(columnMEAN);
71  graph->setPercentile75Column(columnQ75);
72  graph->setMaxColumn(columnMAX);
73  graph->setTitle("vertical Boxplots");
74 
75  // 5. outliers need to be drawn separately
76  JKQTPXYLineGraph* graphOutliers=new JKQTPXYLineGraph(&plot);
77  graphOutliers->setXColumn(columnOUTLIERSX);
78  graphOutliers->setYColumn(columnOUTLIERSY);
79  graphOutliers->setTitle("outliers");
80  // make the color a darker shade of the color of graph
81  graphOutliers->setColor(graph->getColor().darker());
82  graphOutliers->setFillColor(QColor("white"));
83  // draw outliers as small circles, without lines
84  graphOutliers->setSymbol(JKQTPCircle);
85  graphOutliers->setDrawLine(false);
86  graphOutliers->setSymbolSize(7);
87  \endcode
